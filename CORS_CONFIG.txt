import os
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

# Get allowed origins from environment, default to localhost for dev
ALLOWED_ORIGINS = os.getenv(
    "ALLOWED_ORIGINS",
    "http://localhost:3000,http://127.0.0.1:3000,http://127.0.0.1:3003"
).split(",")

app = FastAPI(title="NDAY OM API")

# Configure CORS for production and local development
app.add_middleware(
    CORSMiddleware,
    allow_origins=ALLOWED_ORIGINS,
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)
